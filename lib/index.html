Recursion and conditionals

      
(def do
  (lambda (a b)
    b))

(def count-to-10
  (lambda (c)
    (do (println c)
        (if (= c 10)
          (quote worked)
          (count-to-10 (+ c 1))))))

(count-to-10 0)
